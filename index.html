    JScrollPane scrollPane = new JScrollPane();
    scrollPane.setBounds(0, 0, 1013, 534);
    contentPane.add(scrollPane);

    JMenuBar menuBar = new JMenuBar();
    scrollPane.setColumnHeaderView(menuBar);

    JLabel lblNewLabel = new JLabel("URL:");
    lblNewLabel.setFont(new Font("宋体", Font.PLAIN, 18));
    lblNewLabel.setBackground(Color.WHITE);
    menuBar.add(lblNewLabel);

    JTextArea text_url = new JTextArea();
    text_url.setFont(new Font("Monospaced", Font.PLAIN, 18));
    text_url.setBackground(Color.WHITE);
    menuBar.add(text_url);

    JButton btnNewButton = new JButton("\u83B7\u53D6\u6E90\u7801");
    btnNewButton.setFont(new Font("宋体", Font.PLAIN, 18));
    btnNewButton.setBackground(Color.WHITE);
    btnNewButton.addActionListener(new ActionListener() {
        public void actionPerformed(ActionEvent e) {
            txt.setText(null);
            try {
                URL url = new URL(text_url.getText());
                try{
                InputStream is =url.openStream();
                InputStreamReader isr = new InputStreamReader(is,"UTF-8");
                BufferedReader bf = new BufferedReader(isr);
                String s=bf.readLine();
                while(s!=null)
                {
                    txt.append(s+"\n");
                    s=bf.readLine();
                }
                bf.close();
                isr.close();
                is.close();
                }
            catch (IOException e1) {
            } 
            }catch (MalformedURLException e2) {
            }

        }
    });
    menuBar.add(btnNewButton);

    JButton button = new JButton("\u4FDD\u5B58\u6587\u4EF6");
    button.setFont(new Font("宋体", Font.PLAIN, 18));
    button.setBackground(Color.WHITE);
    button.addActionListener(new ActionListener() {
        public void actionPerformed(ActionEvent e) {
            try{
                JFrame jf = new JFrame();
                FileDialog fd = new FileDialog(jf,"保存",FileDialog.SAVE);
                fd.setVisible(true);
                String filepath = fd.getDirectory();
                String filename = fd.getFile();
                try{
                    FileOutputStream fos = new FileOutputStream(filepath+filename);
                    String s;
                    s=txt.getText();
                    fos.write(s.getBytes());
                    fos.close();
                }catch (Exception ex) {
                }
            }catch (Exception ex) {
            }

        }
    });
    menuBar.add(button);

    JButton btnNewButton_1 = new JButton("\u83B7\u53D6\u56FE\u7247");

                try{
                    InputStream is = url.openStream();
                    File out = new File("d:\\pic8.jpg");
                    FileOutputStream fos = new FileOutputStream(out);
                    byte[] buff = new byte[1024];
                    int n = is.read(buff);
                    while(n!=-1)
                    {
                        fos.write(buff,0,n);
                        n = is.read(buff);
                    }
                    fos.close();
                    is.close();
                }
